
import pandas as pd

ruta1=input("importar datos")
print(ruta1)
ruta2=input("importemos la segunta hoja de calculo")
print(ruta2)
df1=pd.read_excel(ruta1)
df2=pd.read_excel(ruta2)
df1['Datos personales o de empresa'] = df1['Datos personales o de empresa'].astype(str).str.strip()
df2['Datos personales o de empresa'] = df2['Datos personales o de empresa'].astype(str).str.strip()
df_merge=pd.merge(df1,df2, on="Datos personales o de empresa",how="inner")
print("\nContenido de df_merge (primeras 5 filas):")
print(df_merge.head())
print("\nColumnas de df_merge:")
print(df_merge.columns)
print("\nValores únicos en la columna 'Datos personales o de empresa':")
print(df_merge['Datos personales o de empresa'].unique())
print("\nValores únicos en la columna 'Estado_x' de df_merge:")
print(df_merge['Estado_x'].unique())
estado_a_filtrar = input("Ingresa el estado por el que deseas filtrar (ej: 'Entregado'): ")
df_filtrado_por_estado = df_merge[df_merge['Estado_x'] == estado_a_filtrar]
print(f"\nContenido de df_merge filtrado por '{estado_a_filtrar}' (primeras 5 filas):")
print(df_filtrado_por_estado.head())
print("\nColumnas de df_merge filtrado:")
print(df_filtrado_por_estado.columns)
print("\nValores únicos en la columna 'SKU_x' de df_merge:")
print(df_merge['SKU_x'].unique())
print("\nColumnas de df_merge:")
print(df_merge.columns)
print("\nValores únicos en la columna 'Unidades_x' de df_merge:")
print(df_merge['Unidades_x'].unique())
unidades_a_filtrar = int(input("Ingresa el número de unidades por el que deseas filtrar (ej: '1'): "))
df_filtrado_por_unidades = df_merge[df_merge['Unidades_x'] == unidades_a_filtrar]
print(f"\nContenido de df_merge filtrado por Unidades '{unidades_a_filtrar}' (primeras 5 filas):")
print(df_filtrado_por_unidades.head())
print("\nColumnas de df_merge filtrado por Unidades:")
print(df_filtrado_por_unidades.columns)
